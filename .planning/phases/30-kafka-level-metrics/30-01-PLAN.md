/// Returns a snapshot of committed offsets for all registered topic-partitions.
///
/// Used by KafkaMetricsTask to report committed_offset gauge per TP.
/// Returns empty map if no partitions are registered.
pub fn offset_snapshots(&self) -> std::collections::HashMap<(String, i32), i64> {
    let guard = self.partitions.lock();
    guard
        .iter()
        .map(|(key, state)| {
            let TopicPartitionKey(t, p) = key;
            ((t.clone(), *p), state.committed_offset)
        })
        .collect()
}

    poll_interval: std::time::Duration,
    shutdown_rx: tokio::sync::watch::Receiver&lt;()&gt;,
}

impl KafkaMetricsTask {
    pub fn spawn(
        runner: std::sync::Arc&lt;ConsumerRunner&gt;,
        offset_tracker: std::sync::Arc&lt;OffsetTracker&gt;,
        metrics: std::sync::Arc&lt;KafkaMetrics&gt;,
        poll_interval: std::time::Duration,
    ) -> Self {
        let (shutdown_tx, shutdown_rx) = tokio::sync::watch::channel(());
        // Spawn background task
        let task = Self { /* ... */ };
        tokio::spawn(task.run());
        task
    }
    
    async fn run(self) {
        let mut interval = tokio::time::interval(self.poll_interval);
        loop {
            tokio::select! {
                _ = interval.tick() => {
                    self.poll_and_update().await;

                _ = self.shutdown_rx.recv() => {
                    break;
                }

        }
