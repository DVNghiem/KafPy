```rust
use std::sync::atomic::{AtomicUsize, Ordering};
use std::sync::Arc;
use tokio::sync::mpsc;

// HandlerMetadata holds atomic counters for queue_depth and inflight
// - queue_depth: messages in the bounded channel buffer (sender-side)
 // - inflight: messages dispatched but not yet acknowledged via ack()
pub struct HandlerMetadata {
    pub capacity: usize,
    pub queue_depth: AtomicUsize,
    pub inflight: AtomicUsize,
}

impl HandlerMetadata {
    pub fn new(capacity: usize) -> Self {
        Self {
            capacity,
            queue_depth: AtomicUsize::new(0),
            inflight: AtomicUsize::new(0),
        }
    }
}

// HandlerEntry pairs the mpsc::Sender with its metadata
pub struct HandlerEntry {
    pub sender: mpsc::Sender<OwnedMessage>,
    pub metadata: HandlerMetadata,
}

// QueueManager owns all handler queues and metadata.
// Thread-safe via parking_lot::Mutex.
pub struct QueueManager {
    handlers: parking_lot::Mutex<std::collections::HashMap<String, HandlerEntry>>,
}

impl QueueManager {
    pub fn new() -> Self;
    
    // Registers handler for topic with given capacity.
    // Returns mpsc::Receiver for the Python/PyO3 layer.
    // Thread-safe.
    pub fn register_handler(
        &self,
        topic: impl Into<String>,
        capacity: usize,
    ) -> mpsc::Receiver<OwnedMessage>;
    
    // Get queue depth for topic. Returns None if topic not registered.
    pub fn get_queue_depth(&self, topic: &str) -> Option<usize>;
    
    // Get inflight count for topic. Returns None if topic not registered.
    pub fn get_inflight(&self, topic: &str) -> Option<usize>;
    
    // Called by Python/PyO3 layer after processing N messages.
    // Decrements both queue_depth and inflight by count.
    // Idempotent - no-op if topic not found.
    pub fn ack(&self, topic: &str, count: usize);
    
    // Internal: send message to topic's handler, incrementing counters.
    // Returns DispatchOutcome on success, DispatchError on failure.
    pub(crate) fn send_to_handler(
        &self,
        message: OwnedMessage,
    ) -> Result<DispatchOutcome, DispatchError>;
}
```

IMPORTANT IMPLEMENTATION DETAILS:
- `send_to_handler` uses `try_send()` on the sender. On `Full` error, return `DispatchError::QueueFull(topic)`. On `Closed` error, return `DispatchError::QueueClosed(topic)`.
- On successful `try_send`, increment `queue_depth` by 1 (message is now buffered).
- Increment `inflight` by 1 immediately on dispatch (message dispatched but not yet processed).
- `ack(topic, count)` decrements both counters by count. Use `fetch_sub`.
- Use `Ordering::Relaxed` for all atomic operations (no synchronization guarantees needed between threads for this use case).
- Do NOT implement Default for QueueManager — use `new()` explicitly.
- Module should be `pub(crate)` for internal use; queue_manager is internal infrastructure.
