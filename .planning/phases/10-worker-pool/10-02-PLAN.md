pub fn stop(&self) {
    let mut pool_guard = self.pool.lock().unwrap();
    if let Some(pool) = pool_guard.take() {
        // Note: shutdown() is async, can't await in sync stop()
        // Solution: spawn a task to shutdown (stop is fire-and-forget signal)
        tokio::spawn(async move {
            pool.shutdown().await;
        });
    }

**Important note on stop()**: Since `stop()` is called synchronously from Python but `shutdown()` is async, we use `tokio::spawn` to drive the async shutdown.
