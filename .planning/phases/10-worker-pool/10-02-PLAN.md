```rust
pub fn start(&self, py: Python<'_>) -> PyResult<Py<PyAny>> {
    let config = self.config.clone();
    let handlers = Arc::clone(&self.handlers);

    pyo3_async_runtimes::tokio::future_into_py(py, async move {
        let rust_config = /* same as existing */;
        let runner = ConsumerRunner::new(rust_config)
            .map_err(|e| PyErr::new::<pyo3::exceptions::PyRuntimeError, _>(e.to_string()))?;

        // Create ConsumerDispatcher (owned, not stored in Consumer struct)
        let dispatcher = ConsumerDispatcher::new(runner);

        // Collect receivers from all registered handlers
        let receivers: Vec<_> = {
            let handlers_guard = handlers.lock().unwrap();
            handlers_guard
                .keys()
                .map(|topic| dispatcher.register_handler(topic.clone(), 100, None))
                .collect()
        };

        // Build PythonHandler from the registered callbacks
        let handler_arc: Arc<PythonHandler> = {
            let handlers_guard = handlers.lock().unwrap();
            let first_handler = handlers_guard.values().next()
                .cloned()
                .expect("at least one handler must be registered");
            Arc::new(PythonHandler::new(first_handler))
        };

        let executor_arc: Arc<dyn Executor> = Arc::new(DefaultExecutor::default());
        let queue_manager_arc = dispatcher.dispatcher().queue_manager();

        let n_workers = 4; // EXEC-08: configurable

        let pool = WorkerPool::new(
            n_workers,
            receivers,
            handler_arc,
            executor_arc,
            queue_manager_arc.clone(),
        );

        // Store pool for stop() to call shutdown()
        {
            let mut pool_guard = self.pool.lock().unwrap();
            *pool_guard = Some(pool);
        }

        // CRITICAL FIX (Issue 2): Run dispatcher.run() and pool.run() concurrently.
        // Use tokio::spawn so both run in parallel. pool.run() awaits join_set.shutdown()
        // which completes when stop() cancels the CancellationToken.
        tokio::spawn(async move {
            dispatcher.run(&DefaultBackpressurePolicy).await;
        });
        pool.run().await;

        Ok(())
    })
    .map(|b| b.unbind())
}
```

**6. Modify `stop()`** — cancel the CancellationToken by calling pool.shutdown():
```rust
pub fn stop(&self) {
    let mut pool_guard = self.pool.lock().unwrap();
    if let Some(mut pool) = pool_guard.take() {
        // spawn shutdown task (stop() is sync but shutdown() is async)
        tokio::spawn(async move {
            pool.shutdown().await;
        });
    }
}
```

**Note**: The `WorkerPool::shutdown()` method calls `self.shutdown_token.cancel()` which is detected by all worker loops via `tokio::select!` on `shutdown_token.cancelled()`. Workers finish their in-flight message before exiting (EXEC-12).
