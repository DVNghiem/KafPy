//! Unit tests for the dispatcher module.

use kafpy::dispatcher::{Dispatcher, DispatchOutcome, DispatchError};
use kafpy::consumer::OwnedMessage;
use kafpy::consumer::MessageTimestamp;

fn make_message(topic: &str, partition: i32, offset: i64) -> OwnedMessage {
    OwnedMessage {
        topic: topic.to_string(),
        partition,
        offset,
        key: None,
        payload: Some(b"test".to_vec()),
        timestamp: MessageTimestamp::NotAvailable,
        headers: vec![],
    }
}

#[test]
fn dispatcher_new_creates_empty_handlers() {
    let dispatcher = Dispatcher::new();
    // Cannot directly check internal state, but send to unregistered topic should fail
    let msg = make_message("unknown-topic", 0, 0);
    let result = dispatcher.send(msg);
    assert!(result.is_err());
}

#[test]
fn register_handler_returns_receiver() {
    let dispatcher = Dispatcher::new();
    let rx = dispatcher.register_handler("test-topic", 10);
    // Receiver should be usable - drop it immediately
    drop(rx);
}

#[test]
fn register_handler_allows_sending_after_registration() {
    let dispatcher = Dispatcher::new();
    let _rx = dispatcher.register_handler("test-topic", 10);
    let msg = make_message("test-topic", 0, 1);
    let result = dispatcher.send(msg);
    assert!(result.is_ok());
}

#[test]
fn send_returns_dispatch_outcome_on_success() {
    let dispatcher = Dispatcher::new();
    let _rx = dispatcher.register_handler("test-topic", 10);
    let msg = make_message("test-topic", 2, 100);
    let result = dispatcher.send(msg);
    assert!(result.is_ok());
    let outcome = result.unwrap();
    assert_eq!(outcome.topic, "test-topic");
    assert_eq!(outcome.partition, 2);
    assert_eq!(outcome.offset, 100);
}

#[test]
fn send_returns_handler_not_registered_for_unknown_topic() {
    let dispatcher = Dispatcher::new();
    let _rx = dispatcher.register_handler("other-topic", 10);
    let msg = make_message("unknown-topic", 0, 1);
    let result = dispatcher.send(msg);
    assert!(result.is_err());
    let err = result.unwrap_err();
    assert!(matches!(err, DispatchError::HandlerNotRegistered(_)));
}

#[test]
fn send_returns_queue_full_when_channel_full() {
    let dispatcher = Dispatcher::new();
    // Register with capacity 1
    let rx = dispatcher.register_handler("test-topic", 1);

    // Fill the queue
    let msg1 = make_message("test-topic", 0, 1);
    dispatcher.send(msg1).expect("first send should succeed");

    // Now the queue is full - try to send another
    let msg2 = make_message("test-topic", 0, 2);
    let result = dispatcher.send(msg2);
    assert!(result.is_err());
    let err = result.unwrap_err();
    assert!(matches!(err, DispatchError::QueueFull(_)));

    drop(rx);
}

#[test]
fn send_returns_queue_closed_when_receiver_dropped() {
    let dispatcher = Dispatcher::new();
    let rx = dispatcher.register_handler("test-topic", 10);
    drop(rx); // Close the handler's receiver

    let msg = make_message("test-topic", 0, 1);
    let result = dispatcher.send(msg);
    assert!(result.is_err());
    let err = result.unwrap_err();
    assert!(matches!(err, DispatchError::QueueClosed(_)));
}

#[test]
fn dispatch_error_display_trait() {
    let dispatcher = Dispatcher::new();
    let _rx = dispatcher.register_handler("test-topic", 10);

    // Test HandlerNotRegistered error message format
    let msg = make_message("unknown", 0, 1);
    let result = dispatcher.send(msg);
    let err = result.unwrap_err();
    let msg_str = format!("{}", err);
    assert!(msg_str.contains("unknown"));
    assert!(msg_str.contains("HandlerNotRegistered") || msg_str.contains("no handler registered"));
}

#[test]
fn dispatch_error_debug_trait() {
    let dispatcher = Dispatcher::new();
    let _rx = dispatcher.register_handler("test-topic", 10);

    let msg = make_message("unknown", 0, 1);
    let result = dispatcher.send(msg);
    let err = result.unwrap_err();
    let debug_str = format!("{:?}", err);
    assert!(!debug_str.is_empty());
}

#[test]
fn dispatch_outcome_clone() {
    let outcome = DispatchOutcome {
        topic: "test".to_string(),
        partition: 0,
        offset: 100,
        queue_depth: 5,
    };
    let cloned = outcome.clone();
    assert_eq!(cloned.topic, outcome.topic);
    assert_eq!(cloned.partition, outcome.partition);
    assert_eq!(cloned.offset, outcome.offset);
    assert_eq!(cloned.queue_depth, outcome.queue_depth);
}

#[test]
fn multiple_topics_independent_queues() {
    let dispatcher = Dispatcher::new();
    let _rx1 = dispatcher.register_handler("topic-a", 10);
    let _rx2 = dispatcher.register_handler("topic-b", 10);

    let msg_a = make_message("topic-a", 0, 1);
    let msg_b = make_message("topic-b", 1, 2);

    let result_a = dispatcher.send(msg_a);
    let result_b = dispatcher.send(msg_b);

    assert!(result_a.is_ok());
    assert!(result_b.is_ok());
}

#[test]
fn dispatch_error_unknown_topic_variant_exists() {
    // Verify UnknownTopic variant exists (DISP-19 requirement)
    // UnknownTopic = topic doesn't exist in Kafka cluster (not subscribed)
    let err = DispatchError::UnknownTopic("my-topic".to_string());
    let msg_str = format!("{}", err);
    let debug_str = format!("{:?}", err);
    assert!(!msg_str.is_empty());
    assert!(!debug_str.is_empty());
}

#[test]
fn dispatch_error_queue_full_variant_exists() {
    // Verify QueueFull variant exists (DISP-19 requirement)
    let err = DispatchError::QueueFull("my-topic".to_string());
    let msg_str = format!("{}", err);
    let debug_str = format!("{:?}", err);
    assert!(msg_str.contains("full"));
    assert!(!debug_str.is_empty());
}

#[test]
fn dispatch_error_handler_not_registered_variant_exists() {
    // Verify HandlerNotRegistered variant exists (DISP-19 requirement)
    // HandlerNotRegistered = topic exists but no Python handler is registered
    let err = DispatchError::HandlerNotRegistered("my-topic".to_string());
    let msg_str = format!("{}", err);
    let debug_str = format!("{:?}", err);
    assert!(msg_str.contains("handler") || msg_str.contains("registered"));
    assert!(!debug_str.is_empty());
}

#[test]
fn dispatch_error_queue_closed_variant_exists() {
    // Verify QueueClosed variant exists (DISP-19 requirement)
    let err = DispatchError::QueueClosed("my-topic".to_string());
    let msg_str = format!("{}", err);
    let debug_str = format!("{:?}", err);
    assert!(msg_str.contains("closed"));
    assert!(!debug_str.is_empty());
}
