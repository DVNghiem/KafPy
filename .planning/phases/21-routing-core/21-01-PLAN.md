//! RoutingDecision — outcome of a routing evaluation.

use crate::routing::HandlerId;

/// Rejection reason for RoutingDecision::Reject.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum RejectReason {
    /// No matching routing rule found.
    NoMatch,
    /// No default handler configured in the routing chain.
    NoDefaultHandler,
    /// Explicit rejection by a router.
    Explicit(String),
}

impl std::fmt::Display for RejectReason {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            RejectReason::NoMatch => write!(f, "no matching routing rule"),
            RejectReason::NoDefaultHandler => write!(f, "no default handler configured"),
            RejectReason::Explicit(s) => write!(f, "rejected: {s}"),
        }
    }
}

/// Result of evaluating a message against a router.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum RoutingDecision {
    /// Route to the given handler ID.
    Route(HandlerId),
    /// Drop the message silently (offset advances, no handler invoked, no DLQ).
    Drop,
    /// Reject the message with a reason (routes to DLQ in Phase 23).
    Reject(RejectReason),
    /// Defer to the next router in the chain (no match for this router).
    Defer,
}

impl RoutingDecision {
    /// Returns the handler ID if this is a Route decision.
    pub fn as_route(&self) -> Option<&HandlerId> {
        match self {
            RoutingDecision::Route(id) => Some(id),
            _ => None,
        }
    }

    /// Returns true if this is a Drop decision.
    pub fn is_drop(&self) -> bool {
        matches!(self, RoutingDecision::Drop)
    }

    /// Returns true if this is a Defer decision.
    pub fn is_defer(&self) -> bool {
        matches!(self, RoutingDecision::Defer)
    }
}

impl Default for RoutingDecision {
    /// Default decision is Defer — continue to next router.
    fn default() -> Self {
        RoutingDecision::Defer
    }
}

//! RoutingContext — zero-copy borrowed view of a message for routing evaluation.
//!
//! Borrows from `OwnedMessage` without copying any data. The lifetime `'a`
//! is tied to the source message, ensuring no use-after-free.

use crate::consumer::message::OwnedMessage;

/// Handler identifier — a string uniquely identifying a registered handler.
///
/// Phase 23 wires this to per-handler mpsc channels in ConsumerDispatcher.
pub type HandlerId = String;

/// Zero-copy routing context — borrows topic, headers, key, and payload from the message.
///
/// Constructed via `RoutingContext::from_message(&msg)` which does NOT heap-allocate.
/// Headers are iterated in-place from the owned `Vec<(String, Option<Vec<u8>>)>`.
#[derive(Debug, Clone, Copy)]
pub struct RoutingContext<'a> {
    /// Topic name — borrowed from OwnedMessage::topic.
    pub topic: &'a str,
    /// Topic partition.
    pub partition: i32,
    /// Topic offset.
    pub offset: i64,
    /// Message key — borrowed from OwnedMessage::key (None if absent).
    pub key: Option<&'a [u8]>,
    /// Message payload — borrowed from OwnedMessage::payload (None if absent).
    pub payload: Option<&'a [u8]>,
    /// Message headers — borrowed reference to the owned headers Vec.
    /// Each header is (key, Some(value) | None).
    pub headers: &'a [(String, Option<Vec<u8>>)],
}

impl<'a> RoutingContext<'a> {
    /// Constructs a RoutingContext by borrowing from an OwnedMessage.
    ///
    /// No heap allocation. No data copied.
    pub fn from_message(msg: &'a OwnedMessage) -> Self {
        Self {
            topic: &msg.topic,
            partition: msg.partition,
            offset: msg.offset,
            key: msg.key.as_deref(),
            payload: msg.payload.as_deref(),
            headers: &msg.headers,
        }
    }

    /// Returns the payload as a UTF-8 string, if valid.
    pub fn payload_str(&self) -> Option<&'a str> {
        self.payload.and_then(|b| std::str::from_utf8(b).ok())
    }

    /// Returns the key as a UTF-8 string, if valid.
    pub fn key_str(&self) -> Option<&'a str> {
        self.key.and_then(|k| std::str::from_utf8(k).ok())
    }

    /// Iterates over headers, yielding (key, value) pairs where value is None for null headers.
    pub fn headers_iter(&self) -> impl Iterator<Item = (&str, Option<&[u8]>)> + 'a {
        self.headers.iter().map(|(k, v)| (k.as_str(), v.as_deref()))
    }

    /// Returns true if a header with the given key exists.
    pub fn has_header(&self, key: &str) -> bool {
        self.headers.iter().any(|(k, _)| k == key)
    }

    /// Returns the value of the first header with the given key, if present.
    pub fn get_header(&self, key: &str) -> Option<&[u8]> {
        self.headers
            .iter()
            .find(|(k, _)| k == key)
            .and_then(|(_, v)| v.as_deref())
    }
}

//! Router trait — implemented by all routing strategies.
//!
//! Mirrors the `DlqRouter` trait pattern: simple interface, Send + Sync bounds,
//! boxed as `Box<dyn Router>` in RoutingChain.

use crate::routing::context::RoutingContext;
use crate::routing::decision::RoutingDecision;

/// Trait for routing a message to a handler.
///
/// Implement this trait for custom routing strategies. All implementations
/// must be safe to share across threads (Send + Sync).
///
/// # Example
/// ```ignore
/// struct MyRouter;
/// impl Router for MyRouter {
///     fn route(&self, ctx: &RoutingContext) -> RoutingDecision {
///         if ctx.topic.starts_with("events.") {
///             RoutingDecision::Route("events-handler".into())
///         } else {
///             RoutingDecision::Defer
///         }
///     }
/// }
/// ```
pub trait Router: Send + Sync {
    /// Evaluates the routing context and returns a routing decision.
    ///
    /// Implementations should return `RoutingDecision::Defer` when the
    /// message does not match this router's criteria, allowing the next
    /// router in the chain to evaluate it.
    fn route(&self, ctx: &RoutingContext) -> RoutingDecision;
}
