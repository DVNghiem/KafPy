//! TopicPatternRouter — routes based on topic name glob or regex patterns.
//!
//! Per D-01: both glob and regex supported, user picks via PatternType enum.

use crate::routing::context::RoutingContext;
use crate::routing::decision::RoutingDecision;
use crate::routing::router::Router;
use glob::Pattern;
use regex::Regex;
use std::sync::Arc;

/// Pattern type for topic matching.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum PatternType {
    /// Glob pattern (e.g., "events.*", "*-dev").
    Glob,
    /// Regular expression (e.g., "^events\\..+$").
    Regex,
}

/// A single topic pattern rule with a target handler.
#[derive(Debug, Clone)]
pub struct TopicRule {
    /// The pattern string (glob or regex syntax).
    pub pattern: String,
    /// The type of pattern (glob or regex).
    pub pattern_type: PatternType,
    /// Handler ID to route to if this pattern matches.
    pub handler_id: crate::routing::HandlerId,
}

/// TopicPatternRouter — matches topic names against glob or regex patterns.
///
/// Iterates rules in order; returns first matching Route, or Defer if no match.
///
/// # Example
/// ```ignore
/// let router = TopicPatternRouter::new(vec![
///     TopicRule {
///         pattern: "events.*".into(),
///         pattern_type: PatternType::Glob,
///         handler_id: "events-handler".into(),
///     },
/// ]);
/// ```
#[derive(Debug, Clone)]
pub struct TopicPatternRouter {
    rules: Arc<[TopicRule]>,
    // Compiled patterns cached to avoid re-parsing on every message.
    // Option<Pattern> for glob, Option<Regex> for regex.
    compiled: Arc<[CompiledPattern]>,
}

#[derive(Debug, Clone)]
enum CompiledPattern {
    Glob(Pattern),
    Regex(Regex),
}

impl TopicPatternRouter {
    /// Creates a new TopicPatternRouter from a list of TopicRules.
    ///
    /// Compiles patterns once at construction time.
    pub fn new(rules: Vec<TopicRule>) -> Result<Self, PatternError> {
        let mut compiled = Vec::with_capacity(rules.len());
        for rule in &rules {
            let cp = match rule.pattern_type {
                PatternType::Glob => {
                    CompiledPattern::Glob(Pattern::new(&rule.pattern).map_err(|e| PatternError::InvalidGlob(rule.pattern.clone(), e.to_string()))?)
                }
                PatternType::Regex => {
                    CompiledPattern::Regex(Regex::new(&rule.pattern).map_err(|e| PatternError::InvalidRegex(rule.pattern.clone(), e.to_string()))?)
                }
            };
            compiled.push(cp);
        }
        Ok(Self {
            rules: rules.into(),
            compiled: compiled.into(),
        })
    }

    /// Returns the number of rules.
    pub fn rule_count(&self) -> usize {
        self.rules.len()
    }
}

impl Router for TopicPatternRouter {
    fn route(&self, ctx: &RoutingContext) -> RoutingDecision {
        for (rule, compiled) in self.rules.iter().zip(self.compiled.iter()) {
            let matches = match compiled {
                CompiledPattern::Glob(p) => p.matches(ctx.topic),
                CompiledPattern::Regex(r) => r.is_match(ctx.topic),
            };
            if matches {
                return RoutingDecision::Route(rule.handler_id.clone());
            }
        }
        RoutingDecision::Defer
    }
}

/// Error type for pattern compilation failures.
#[derive(Debug, thiserror::Error)]
pub enum PatternError {
    #[error("invalid glob pattern '{0}': {1}")]
    InvalidGlob(String, String),
    #[error("invalid regex pattern '{0}': {1}")]
    InvalidRegex(String, String),
}

//! HeaderRouter — routes based on message header key presence or value pattern.
//!
//! Routes by checking if a header key exists (any value) or if the header value
//! matches a glob pattern.

use crate::routing::context::RoutingContext;
use crate::routing::decision::RoutingDecision;
use crate::routing::router::Router;
use glob::Pattern;
use std::sync::Arc;

/// Configuration for a single header routing rule.
#[derive(Debug, Clone)]
pub struct HeaderRule {
    /// Header key to match.
    pub key: String,
    /// Optional glob pattern the header value must match (None = any value is OK).
    pub value_pattern: Option<String>,
    /// Handler ID to route to if this rule matches.
    pub handler_id: crate::routing::HandlerId,
}

/// HeaderRouter — routes by header key presence or value glob pattern.
///
/// Checks each rule in order; first matching rule returns Route, else Defer.
///
/// # Example
/// ```ignore
/// let router = HeaderRouter::new(vec![
///     HeaderRule {
///         key: "content-type".into(),
///         value_pattern: Some("application/json*".into()),
///         handler_id: "json-handler".into(),
///     },
///     HeaderRule {
///         key: "x-trace-id".into(),
///         value_pattern: None,  // any value, just check presence
///         handler_id: "traced-handler".into(),
///     },
/// ]);
/// ```
#[derive(Debug, Clone)]
pub struct HeaderRouter {
    rules: Arc<[HeaderRule]>,
}

impl HeaderRouter {
    /// Creates a new HeaderRouter.
    pub fn new(rules: Vec<HeaderRule>) -> Self {
        Self {
            rules: rules.into(),
        }
    }

    /// Returns the number of rules.
    pub fn rule_count(&self) -> usize {
        self.rules.len()
    }
}

impl Router for HeaderRouter {
    fn route(&self, ctx: &RoutingContext) -> RoutingDecision {
        for rule in self.rules.iter() {
            // Check if header key exists
            if !ctx.has_header(&rule.key) {
                continue;
            }
            // If no value pattern required, presence is enough
            let Some(ref value_pattern) = rule.value_pattern else {
                return RoutingDecision::Route(rule.handler_id.clone());
            };
            // Get header value and match against glob pattern
            if let Some(value) = ctx.get_header(&rule.key) {
                if let Ok(pattern) = Pattern::new(value_pattern) {
                    if let Some(value_str) = std::str::from_utf8(value).ok() {
                        if pattern.matches(value_str) {
                            return RoutingDecision::Route(rule.handler_id.clone());
                        }
                    }
                }
            }
        }
        RoutingDecision::Defer
    }
}

//! KeyRouter — routes based on message key prefix or exact match.
//!
//! Supports both binary key matching (prefix/exact) and UTF-8 string matching.

use crate::routing::context::RoutingContext;
use crate::routing::decision::RoutingDecision;
use crate::routing::router::Router;
use std::sync::Arc;

/// Key match mode.
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum KeyMatchMode {
    /// Exact binary match (full key must equal the pattern).
    Exact,
    /// Key must start with the pattern bytes (prefix match).
    Prefix,
    /// Key must start with the pattern as a UTF-8 string.
    PrefixStr,
    /// Key must exactly equal the pattern as a UTF-8 string.
    ExactStr,
}

/// Configuration for a single key routing rule.
#[derive(Debug, Clone)]
pub struct KeyRule {
    /// Match mode (exact binary, prefix binary, exact string, prefix string).
    pub match_mode: KeyMatchMode,
    /// Pattern bytes or string to match against.
    pub pattern: Vec<u8>,
    /// Handler ID to route to if this rule matches.
    pub handler_id: crate::routing::HandlerId,
}

impl KeyRule {
    /// Creates a binary exact-match rule.
    pub fn exact_bytes(pattern: impl Into<Vec<u8>>, handler_id: crate::routing::HandlerId) -> Self {
        Self { match_mode: KeyMatchMode::Exact, pattern: pattern.into(), handler_id }
    }

    /// Creates a binary prefix-match rule.
    pub fn prefix_bytes(pattern: impl Into<Vec<u8>>, handler_id: crate::routing::HandlerId) -> Self {
        Self { match_mode: KeyMatchMode::Prefix, pattern: pattern.into(), handler_id }
    }

    /// Creates a UTF-8 exact-match rule.
    pub fn exact_str(pattern: impl Into<String>, handler_id: crate::routing::HandlerId) -> Self {
        Self { match_mode: KeyMatchMode::ExactStr, pattern: pattern.into().into_bytes(), handler_id }
    }

    /// Creates a UTF-8 prefix-match rule.
    pub fn prefix_str(pattern: impl Into<String>, handler_id: crate::routing::HandlerId) -> Self {
        Self { match_mode: KeyMatchMode::PrefixStr, pattern: pattern.into().into_bytes(), handler_id }
    }
}

/// KeyRouter — routes by message key prefix or exact match.
///
/// Iterates rules in order; first match returns Route, else Defer.
///
/// # Example
/// ```ignore
/// let router = KeyRouter::new(vec![
///     KeyRule::prefix_str("order-", "orders-handler".into()),
///     KeyRule::exact_str("event-shipping".into(), "shipping-handler".into()),
/// ]);
/// ```
#[derive(Debug, Clone)]
pub struct KeyRouter {
    rules: Arc<[KeyRule]>,
}

impl KeyRouter {
    /// Creates a new KeyRouter.
    pub fn new(rules: Vec<KeyRule>) -> Self {
        Self { rules: rules.into() }
    }

    /// Returns the number of rules.
    pub fn rule_count(&self) -> usize {
        self.rules.len()
    }
}

impl Router for KeyRouter {
    fn route(&self, ctx: &RoutingContext) -> RoutingDecision {
        let Some(key) = ctx.key else {
            return RoutingDecision::Defer;
        };

        for rule in self.rules.iter() {
            let matches = match rule.match_mode {
                KeyMatchMode::Exact => key == rule.pattern.as_slice(),
                KeyMatchMode::Prefix => key.starts_with(&rule.pattern),
                KeyMatchMode::ExactStr => {
                    std::str::from_utf8(key).ok().map_or(false, |s| s.as_bytes() == rule.pattern.as_slice())
                }
                KeyMatchMode::PrefixStr => {
                    std::str::from_utf8(key).ok().map_or(false, |s| s.starts_with(std::str::from_utf8(&rule.pattern).unwrap_or("")))
                }
            };
            if matches {
                return RoutingDecision::Route(rule.handler_id.clone());
            }
        }
        RoutingDecision::Defer
    }
}
