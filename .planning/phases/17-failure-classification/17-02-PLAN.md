**Step B: Create `src/failure/logging.rs`**

Create this new file with exact contents:
```rust
use crate::python::ExecutionContext;
use crate::failure::{FailureReason, FailureCategory};

/// Log a failure with structured context at appropriate level
pub fn log_failure(context: &ExecutionContext, reason: &FailureReason, exception_name: &str, is_retry: bool) {
    match reason.category() {
        FailureCategory::Retryable => {
            tracing::warn!(
                target: "kafpy::failure",
                topic = %context.topic,
                partition = context.partition,
                offset = context.offset,
                reason = %reason,
                exception = %exception_name,
                is_retry = is_retry,
                "Retryable failure in handler"
            );
        }
        FailureCategory::Terminal => {
            tracing::error!(
                target: "kafpy::failure",
                topic = %context.topic,
                partition = context.partition,
                offset = context.offset,
                reason = %reason,
                exception = %exception_name,
                "Terminal failure — will not be retried"
            );
        }
        FailureCategory::NonRetryable => {
            tracing::info!(
                target: "kafpy::failure",
                topic = %context.topic,
                partition = context.partition,
                offset = context.offset,
                reason = %reason,
                exception = %exception_name,
                "Non-retryable failure"
            );
        }
    }
}
```

```rust
fn mark_failed(&self, topic: &str, partition: i32, offset: i64, reason: &FailureReason) {
    let mut state = self.state.lock();
    let tp = TopicPartition::new(topic, partition);
    let ps = state.partitions.entry(tp).or_default();

    ps.pending_offsets.remove(&offset);
    ps.failed_offsets.insert(offset);
    ps.last_failure_reason = Some(reason.clone());

    tracing::debug!(
        topic = %topic,
        partition = partition,
        offset = offset,
        reason = %reason,
        "Marked offset as failed"
    );
}
```
