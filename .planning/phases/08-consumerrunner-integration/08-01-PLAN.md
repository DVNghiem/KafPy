```rust
use rdkafka::consumer::{Consumer, StreamConsumer};
use rdkafka::TopicPartitionList;
use std::collections::{HashMap, HashSet};
use tokio_stream::StreamExt;

// Add after Dispatcher impl block:

/// Owns ConsumerRunner + Dispatcher and orchestrates the async message loop.
/// Wires consumer stream output to dispatcher input.
pub struct ConsumerDispatcher {
    runner: Arc<ConsumerRunner>,
    dispatcher: Dispatcher,
    /// Partition list for each topic, populated via on_assign callback
    partition_handles: parking_lot::Mutex<HashMap&lt;String, TopicPartitionList&gt;&gt;,
    /// Topics currently paused
    paused_topics: parking_lot::Mutex&lt;HashSet&lt;String&gt;&gt;,
    /// Backpressure threshold ratio for resume (0.0 to 1.0)
    resume_threshold: f64,
}

impl ConsumerDispatcher {
    /// Creates a new dispatcher wired to the given runner.
    pub fn new(runner: ConsumerRunner) -> Self {
        Self {
            runner: Arc::new(runner),
            dispatcher: Dispatcher::new(),
            partition_handles: parking_lot::Mutex::new(HashMap::new()),
            paused_topics: parking_lot::Mutex::new(HashSet::new()),
            resume_threshold: 0.5,
        }
    }

    /// Registers a handler for `topic` with bounded queue of `capacity`.
    /// Optionally limits concurrency with `max_concurrency` semaphore permits.
    pub fn register_handler(
        &amp;self,
        topic: impl Into&lt;String&gt;,
        capacity: usize,
        max_concurrency: Option&lt;usize&gt;,
    ) -> mpsc::Receiver&lt;OwnedMessage&gt; {
        let semaphore = max_concurrency.map(|n| Arc::new(tokio::sync::Semaphore::new(n)));
        self.dispatcher.register_handler_with_semaphore(topic, capacity, semaphore)
    }

    /// Runs the dispatch loop, polling the consumer stream and
    /// dispatching each message through the dispatcher.
    /// Uses the provided backpressure policy.
    pub async fn run(&amp;self, policy: &amp;dyn BackpressurePolicy) {
        let mut stream = self.runner.stream();
        while let Some(result) = stream.next().await {
            match result {
                Ok(msg) =&gt; {
                    let topic = msg.topic.clone();
                    match self.dispatcher.send_with_policy(msg, policy) {
                        Ok(outcome) =&gt; {
                            // Check if resume is needed (queue draining below threshold)
                            self.check_resume(&amp;topic, outcome.queue_depth);
                        }
                        Err(DispatchError::Backpressure(topic)) =&gt; {
                            tracing::warn!("backpressure on topic '{}'", topic);
                            // DISP-18: FuturePausePartition signal handled in next plan
                        }
                        Err(DispatchError::HandlerNotRegistered(topic)) =&gt; {
                            tracing::debug!("no handler for topic '{}', skipping", topic);
                        }
                        Err(e) =&gt; {
                            tracing::error!("dispatch error: {}", e);
                        }
                    }
                }
                Err(e) =&gt; {
                    tracing::error!("consumer error: {}", e);
                    // Non-fatal - continue polling
                }
            }
        }
    }

    /// Checks if a paused topic should be resumed based on current queue depth.
    fn check_resume(&amp;self, topic: &amp;str, current_depth: usize) {
        let capacity = match self.dispatcher.get_queue_depth(topic) {
            // queue_depth is current depth, not capacity - use get_capacity instead
            Some(_) =&gt; self.dispatcher.get_capacity(topic),
            None =&gt; None,
        };
        let Some(capacity) = capacity else { return };
        let threshold = (capacity as f64 * self.resume_threshold) as usize;

        if current_depth &lt; threshold {
            if self.paused_topics.lock().remove(topic) {
                tracing::info!("resuming topic '{}' (depth {} &lt; threshold {})", topic, current_depth, threshold);
            }
        }
    }

    /// Returns a reference to the underlying dispatcher for inspection.
    pub fn dispatcher(&amp;self) -> &amp;Dispatcher {
        &amp;self.dispatcher
    }
}
```

Also add `pub fn get_capacity(&amp;self, topic: &amp;str) -> Option&lt;usize&gt;` to `Dispatcher` that delegates to `QueueManager::get_capacity`.

The `run()` method is the core DISP-16 integration - it polls `stream.next()` and dispatches each `OwnedMessage` via `send_with_policy()`.

**Note**: The `FuturePausePartition` handling is stubbed out here - full pause/resume wiring comes in plan 08-02.
