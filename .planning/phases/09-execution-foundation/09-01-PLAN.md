//! Execution result types — normalized outcome of Python handler execution.

use thiserror::Error;

/// Normalized execution result from a Python handler.
///
/// Variants:
/// - `Ok` — handler executed successfully
/// - `Error` — Python exception raised during execution
/// - `Rejected` — handler explicitly rejected the message (invalid result, etc.)
#[derive(Debug, Clone)]
pub enum ExecutionResult {
    /// Handler completed successfully.
    Ok,
    /// Python exception raised during execution.
    Error {
        /// Exception type name (e.g., "ValueError", "RuntimeError").
        exception: String,
        /// Formatted traceback string.
        traceback: String,
    },
    /// Handler explicitly rejected the message (returned invalid result, etc.).
    Rejected {
        /// Human-readable reason for rejection.
        reason: String,
    },
}

impl ExecutionResult {
    /// Returns `true` if this is `Ok`.
    pub fn is_ok(&self) -> bool {
        matches!(self, ExecutionResult::Ok)
    }

    /// Returns `true` if this is `Error`.
    pub fn is_error(&self) -> bool {
        matches!(self, ExecutionResult::Error { .. })
    }

    /// Returns `true` if this is `Rejected`.
    pub fn is_rejected(&self) -> bool {
        matches!(self, ExecutionResult::Rejected { .. })
    }
}

//! Execution context — metadata attached to each message during execution.

/// Context carried through the execution pipeline.
///
/// Provides trace context (topic, partition, offset, worker_id) so that
/// errors, logs, and future ack/retry decisions can reference the
/// original Kafka message coordinates.
#[derive(Debug, Clone)]
pub struct ExecutionContext {
    /// Topic the message was consumed from.
    pub topic: String,
    /// Partition the message was from.
    pub partition: i32,
    /// Offset of the message.
    pub offset: i64,
    /// Worker ID that picked up this message (0-indexed within the WorkerPool).
    pub worker_id: usize,
}

impl ExecutionContext {
    /// Creates a new execution context.
    pub fn new(topic: String, partition: i32, offset: i64, worker_id: usize) -> Self {
        Self {
            topic,
            partition,
            offset,
            worker_id,
        }
    }
}

//! Execution policy — pluggable strategies for handling execution results.

use crate::dispatcher::OwnedMessage;
use crate::python::context::ExecutionContext;
use crate::python::execution_result::ExecutionResult;

/// Outcome of an executor's decision on what to do after execution.
#[derive(Debug, Clone)]
pub enum ExecutorOutcome {
    /// Acknowledge the message — safe to commit offset.
    Ack,
    /// Retry the message — re-dispatch to handler queue.
    Retry,
    /// Reject the message — do not retry, route to DLQ if configured.
    Rejected,
}

/// Policy trait for post-execution decisions.
///
/// Implementors decide what to do after a Python handler runs:
/// - Ack immediately (fire-and-forget)
/// - Retry with backoff
/// - Route to DLQ
///
/// Future implementations: RetryExecutor, CommitExecutor, BatchExecutor.
pub trait Executor: Send + Sync {
    /// Called after a Python handler executes.
    ///
    /// Return `ExecutorOutcome` to signal what the dispatcher should do next.
    fn execute(
        &self,
        ctx: &ExecutionContext,
        _message: &OwnedMessage,
        result: &ExecutionResult,
    ) -> ExecutorOutcome;
}

/// Fire-and-forget executor.
///
/// Logs the outcome at trace level and always returns `Ack`.
/// Used as the default when no retry or commit policy is configured.
#[derive(Debug, Clone, Default)]
pub struct DefaultExecutor;

impl Executor for DefaultExecutor {
    fn execute(
        &self,
        ctx: &ExecutionContext,
        message: &OwnedMessage,
        result: &ExecutionResult,
    ) -> ExecutorOutcome {
        match result {
            ExecutionResult::Ok => {
                tracing::trace!(
                    topic = %ctx.topic,
                    partition = ctx.partition,
                    offset = ctx.offset,
                    worker_id = ctx.worker_id,
                    "handler executed successfully"
                );
            }
            ExecutionResult::Error { exception, traceback } => {
                tracing::warn!(
                    topic = %ctx.topic,
                    partition = ctx.partition,
                    offset = ctx.offset,
                    worker_id = ctx.worker_id,
                    exception = %exception,
                    "handler raised exception"
                );
                // Fire-and-forget: still ack even on error
                // Future RetryExecutor will override this
            }
            ExecutionResult::Rejected { reason } => {
                tracing::warn!(
                    topic = %ctx.topic,
                    partition = ctx.partition,
                    offset = ctx.offset,
                    worker_id = ctx.worker_id,
                    reason = %reason,
                    "handler rejected message"
                );
            }
        }
        ExecutorOutcome::Ack
    }
}
