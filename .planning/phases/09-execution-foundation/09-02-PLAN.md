```rust
//! Python handler — invokes a Python callback via spawn_blocking with minimal GIL window.

use crate::dispatcher::{DispatchError, OwnedMessage};
use crate::python::context::ExecutionContext;
use crate::python::execution_result::ExecutionResult;
use std::sync::Arc;

/// Wraps a Python callable stored as `Py<PyAny>` (GIL-independent, Send+Sync).
///
/// ## PyO3 invariants
///
/// - `Py<PyAny>` is cheap to clone and is Send+Sync when the underlying
///   Python object supports it — this is the required storage form.
/// - NEVER store `&PyAny` or `Bound<'_, PyAny>` — both are lifetime-bound
///   and cannot cross async/thread boundaries.
/// - `callback.unbind()` at the PyO3 boundary converts `Bound` to owned `Py<PyAny>`.
pub struct PythonHandler {
    /// The Python callable — cloned from the Py<PyAny> stored at registration.
    callback: Arc<Py<PyAny>>,
}

impl PythonHandler {
    /// Wraps a Python callable.
    ///
    /// Called by the PyO3 bridge after `callback.unbind()` extracts `Py<PyAny>`.
    pub(crate) fn new(callback: Py<PyAny>) -> Self {
        Self {
            callback: Arc::new(callback),
        }
    }

    /// Invokes the Python callable with the given message.
    ///
    /// Uses `spawn_blocking` to release the Tokio thread during the Python call.
    /// The GIL is acquired only inside the `Python::with_gil` closure.
    pub async fn invoke(
        &self,
        ctx: &ExecutionContext,
        message: OwnedMessage,
    ) -> ExecutionResult {
        let callback = Arc::clone(&self.callback);
        let topic = ctx.topic.clone();
        let partition = ctx.partition;
        let offset = ctx.offset;
        let worker_id = ctx.worker_id;

        // Convert OwnedMessage to py_message (KafkaMessage dict) inside spawn_blocking
        let result = tokio::task::spawn_blocking(move || {
            Python::with_gil(|py| {
                // Build a KafkaMessage dict for Python
                let py_msg = PyDict::new(py);
                py_msg.set_item("topic", &topic).ok();
                py_msg.set_item("partition", partition).ok();
                py_msg.set_item("offset", offset).ok();
                py_msg.set_item("key", message.key.as_deref()).ok();
                py_msg.set_item("payload", message.payload.as_deref()).ok();
                // timestamp: i64 milliseconds
                let ts: i64 = match message.timestamp {
                    MessageTimestamp::NotAvailable => 0,
                    MessageTimestamp::CreateTime(ts) => ts,
                    MessageTimestamp::LogAppendTime(ts) => ts,
                };
                py_msg.set_item("timestamp", ts).ok();
                py_msg.set_item("headers", &message.headers).ok();

                // Invoke the callback
                match callback.call1(py, (py_msg,)) {
                    Ok(_) => ExecutionResult::Ok,
                    Err(py_err) => {
                        let exception = py_err.get_type(py).name().to_string();
                        let traceback = py_err.to_string();
                        ExecutionResult::Error { exception, traceback }
                    }
                }
            })
        })
        .await;

        // Handle spawn_blocking panic — this should not happen but defensively handle
        match result {
            Ok(execution_result) => execution_result,
            Err(_) => ExecutionResult::Error {
                exception: "Panic".to_string(),
                traceback: "spawn_blocking task panicked".to_string(),
            },
        }
    }
}

// Needed for Python::with_gil

use pyo3::types::PyDict;
use crate::consumer::MessageTimestamp;
```

Read src/pyconsumer.rs and add `add_handler(topic: &str, callback: Py<PyAny>)` method to the `Consumer` pyclass. This method:
1. Calls `callback.unbind()` to convert `Bound<'_, PyAny>` to owned `Py<PyAny>`
2. Stores `Py<PyAny>` in the Consumer's handler map
3. Registers the handler with the dispatcher for that topic

The pattern:
```rust
#[pymethods]
impl Consumer {
    /// Registers a Python handler callback for a topic.
    ///
    /// `callback` must be a Python callable that accepts a `KafkaMessage` dict.
    /// The callable is stored as `Py<PyAny>` (GIL-independent, Send+Sync).
    pub fn add_handler(&mut self, topic: String, callback: &PyAny) -> PyResult<()> {
        // Convert Bound<&PyAny> to owned Py<PyAny> at the PyO3 boundary
        let owned_callback = callback.unbind().into_owned();
        self.handlers
            .lock()
            .unwrap()
            .insert(topic.clone(), owned_callback);
        // Register with dispatcher if owned
        if let Some(dispatcher) = &self.dispatcher {
            dispatcher.register_handler(&topic, 100); // default capacity
        }
        Ok(())
    }
}
```
