## RuntimeBuilder interface

```rust
// src/runtime/builder.rs
pub struct RuntimeBuilder {
    config: ConsumerConfig,
    handlers: Arc<Mutex<HashMap<String, Arc<Py<PyAny>>>>>,
    shutdown_token: tokio_util::sync::CancellationToken,
}

impl RuntimeBuilder {
    pub fn new(config: ConsumerConfig, handlers: Arc<Mutex<HashMap<String, Arc<Py<PyAny>>>>>, shutdown_token: tokio_util::sync::CancellationToken) -> Self;

    pub async fn build(self) -> Result<Runtime, PyErr>;

    /// Returns the queue_manager_arc for RuntimeSnapshotTask
    pub fn queue_manager(&self) -> Arc<QueueManager>;

    /// Returns the offset_tracker_arc for RuntimeSnapshotTask
    pub fn offset_tracker(&self) -> Arc<OffsetTracker>;
}

pub struct Runtime {
    pub dispatcher_handle: tokio::task::JoinHandle<()>,
    pub committer_handle: tokio::task::JoinHandle<()>,
    pub pool: WorkerPool,
}
```

```rust
pub fn start(&self, py: Python<'_>) -> PyResult<Py<Any>> {
    let config = self.config.clone();
    let handlers = Arc::clone(&self.handlers);
    let shutdown_token = self.shutdown_token.clone();

    pyo3_async_runtimes::tokio::future_into_py(py, async move {
        let builder = RuntimeBuilder::new(config, handlers, shutdown_token);
        let runtime = builder.build().await.map_err(|e| {
            PyErr::new::<pyo3::exceptions::PyRuntimeError, _>(e.to_string())
        })?;
        runtime.run().await;
        Ok(())
    }).map(|b| b.unbind())
}
```
