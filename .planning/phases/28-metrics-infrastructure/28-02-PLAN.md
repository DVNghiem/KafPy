   pub fn queue_snapshots(&self) -> HashMap<HandlerId, QueueSnapshot> {
       let handlers = self.handlers.lock();
       handlers.iter().map(|(id, meta)| {
           (id.clone(), QueueSnapshot {
               queue_depth: meta.queue_depth.clone(),
               inflight: meta.inflight.clone(),
           })
       }).collect()
   }
