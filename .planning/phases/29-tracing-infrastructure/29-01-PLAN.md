3. **enable_otel_tracing function** (OBS-11, OBS-12):
   ```rust
   pub fn enable_otel_tracing(
       config: &amp;ObservabilityConfig,
   ) -&gt; Result&lt;tracing_subscriber::layer::Layered&lt;_, _,&gt;, SetupError&gt; {
       use tracing_subscriber::{layer::SubscriberExt, util::SubscriberInitExt};

       if config.otlp_endpoint.is_none() {
           // Zero-cost: return a noop layer that does nothing
           return Ok(tracing_subscriber::registry().with(tracing_subscriber::fmt::layer()));
       }

       let endpoint = config.otlp_endpoint.as_ref().unwrap();

       // OTLP exporter
       let exporter = opentelemetry_otlp::SpanExporter::builder()
           .with_tonic()
           .with_endpoint(endpoint)
           .build()
           .map_err(|e| SetupError::Exporter(e.to_string()))?;

       // OpenTelemetry pipeline
       let tracer_provider = opentelemetry_sdk::trace::TracerProvider::builder()
           .with_sampler(opentelemetry_sdk::trace::Sampler::TraceIdRatioBased(
               config.sampling_ratio,
           ))
           .with_resource(opentelemetry_sdk::Resource::new([
               opentelemetry_sdk::ResourceAttribute::from("service.name".to_string()),
               opentelemetry_sdk::ResourceAttribute::from(config.service_name.clone()),
           ]))
           .build();

       let tracer = tracer_provider.tracer("kafpy");

       // W3C propagator (OBS-16)
       let propagator = opentelemetry_propagator_w3c::Propagator::new();

       let telemetry_layer = tracing_opentelemetry::layer()
           .with_tracer(tracer)
           .with_propagator(propagator);

       // Log fmt layer
       let fmt_layer = match config.log_format {
           LogFormat::Json => tracing_subscriber::fmt::layer().with_json(),
           LogFormat::Pretty => tracing_subscriber::fmt::layer().pretty(),
           LogFormat::Simple => tracing_subscriber::fmt::layer(),
       };

       let registry = tracing_subscriber::registry()
           .with(telemetry_layer)
           .with(fmt_layer);

       // KafPy does NOT call set_global_default() — user owns the registry
       // Return LayerHandle for user to install
       Ok(registry)
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| enable_otel_tracing() user call | User passes in OTLP endpoint — untrusted network input; validated at OTLP exporter creation |
| W3C headers at spawn_blocking boundary | Trace context from upstream service — treated as untrusted; W3C spec handles malformed headers gracefully |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-29-01 | I (Information Disclosure) | OTLP endpoint URL | mitigate | Endpoint URL is user-provided config; no PII in span names/fields; audit fields before emission |
| T-29-02 | D (Denial of Service) | OTLP exporter connection | mitigate | Timeout on exporter send; graceful degradation if collector unreachable — tracing is best-effort |
| T-29-03 | T (Tampering) | W3C trace headers | accept | W3C tracecontext is signed/propagated by upstream; span.in_scope() prevents cross-contamination |
</threat_model>
